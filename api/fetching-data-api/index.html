获取数据 API

ajax​

一种获取数据的方法。

通常项目的接口请求都放在了 src/services/ 目录下：

src/services/common.js
import {ajax} from 'concisejs'

export async function login(params) {
  return ajax({
    type: 'post',
    url: '',
    data: params
  })
}

export async function getLocations(params) {
  return ajax({
    type: 'get',
    url: '',
    data: params,
    timeout: 3000 // 设置请求超时时长
  })
}

async function getList (params) {
  return ajax({
    type: 'get',
    url: '',
    cancelEnable: true // 允许取消
  })
}
const getListAjax = getList()
// 取消请求
getListAjax.__cancel__()

ajaxExtend​

可扩展的获取数据的方法。

src/services/config.js
import {ajaxExtend} from 'concisejs'

const ajax = ajaxExtend({
  baseURL: '',
  headers: {},
  timeout: 0,
  responseType: 'json'
})

export default ajax

useAjaxExtend​
