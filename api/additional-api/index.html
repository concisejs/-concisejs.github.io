额外的 API

useCookie#

操作 Cookie 的 Hook 。返回当前用户的 cookie 信息，及操作 cookie 的方法。其中，操作方法包括：

  • getCookie : 获取指定 name 的 cookie 值
  • updateCookie : 更新 cookie
  • removeCookie : 删除指定 name 的 cookie
  • clearCookie : 清除所有 cookie
import {useCookie, useEffect} from 'concisejs'
export default function Example() {
const [cookies, cookieHandler] = useCookie()
const addToken = () => {
cookieHandler.updateCookie('AUTHTOKEN', 'euco94vaifk10j19g5nolh9pnb')
}
const addUserInfo = () => {
cookieHandler.updateCookie('_u', '16|admin', {
long: 1000 * 60 * 60, // Number 类型， cookie 的持续时间，以毫秒计，无默认
path: '/', // String 类型， cookie 的生效路径，默认在根目录下
domain: undefined, // String 类型， cookie 的生效域名，默认为该 cookie 的网页所在的域名
secure: false // Boolean 类型，是否需要 https 发送 cookie ，默认 false
})
}
const deleteToken = () => {
cookieHandler.removeCookie('AUTHTOKEN')
}
const clearAllToken = () => {
cookieHandler.clearCookie()
}
useEffect(() => {
console.log(cookieHandler.getCookie('AUTHTOKEN'))
})
return (
<div>
<div>cookies: {JSON.stringify(cookies)}</div>
<button onClick={addToken}>更新 cookie : AUTHTOKEN=euco94vaifk10j19g5nolh9pnb;</button>
<button onClick={addUserInfo}>更新 cookie : _u=16|admin;</button>
<button onClick={deleteToken}>删除 AUTHTOKEN cookie</button>
<button onClick={clearAllToken}>清除所有 cookie</button>
</div>
)
}

libs#

常用函数集合，包括：

  • sleep : 休眠函数
  • getAllCookies : 获取所有 cookie
import {libs} from 'concisejs'
async function example() {
const res = await libs.sleep(2000, {message: 'ok'})
console.log(res)
}
example() // => {message: 'ok'}
const cookies = libs.getAllCookies()
console.log(cookies)
