路由 API

Link#

链接组件。

包含的属性有：

  • to : String/Object/Function
  • replace : Boolean
  • component : React.Component
  • 其他 <a> 支持的属性，如 id 、 className 等
import {Link, forwardRef} from 'concisejs'
function Example() {
return (
<>
{/* 点击跳转到指定路由 */}
<Link to="/about">About</Link>
{/* 点击跳转到指定路由，并附带 query 参数 */}
<Link to="/courses?sort=name" />
{/*
点击跳转到指定路由，并附带 query 参数、 hash 、 state 参数
这里需要注意的是，当浏览器刷新时， state 参数会丢失
*/}
<Link
to={{
pathname: "/courses",
search: "?sort=name",
hash: "#the-hash",
state: {fromDashboard: true}
}}
/>
{/*
点击跳转到指定路由 /courses ，并附带当前 location 上的所有参数
`to` 支持函数，该函数以当前位置为参数，返回以字符串或对象表示的 location
*/}
<Link to={location => ({...location, pathname: "/courses"})} />
{/* 点击跳转到附带 query ({sort: name}) 的当前路由 */}
<Link to={location => `${location.pathname}?sort=name`} />
{/* `replace` 设置为 true 时，单击链接将替换历史记录堆栈中的当前条目，而不是添加新条目 */}
<Link to="/courses" replace />
{/* 如果想使用自己的导航组件，只需将其通过 `component` 属性传递即可 */}
<Link to="/" component={FancyLink} />
</>
)
}
function FancyLink = forwardRef((props, ref) => (
<a ref={ref} {...props}>fancy link - {props.children}</a>
))

NavLink#

<Link> 的特殊版本，当它与当前 URL 匹配时，可以为元素添加样式。

包含的属性有：

  • activeClassName : String
  • activeStyle : Object
  • exact : Boolean
  • strict : Boolean
  • isActive : Function
import {NavLink} from 'concisejs'
function Example() {
return (
<>
{/* 点击跳转到指定路由，与 Link 等同 */}
<NavLink to="/about">About</NavLink>
{/* 当该链接处于活动状态时，会附加 className selected */}
<NavLink to="/faq" activeClassName="selected">FAQs</NavLink>
{/* 当该链接处于活动状态时，会附加样式 */}
<NavLink
to="/faq"
activeStyle={{
fontWeight: "bold",
color: "red"
}}
>
FAQs
</NavLink>
{/* 当前路由完全匹配为 /profile 时，才会附加 className selected */}
<NavLink exact to="/profile" activeClassName="selected">Profile</NavLink>
{/* 当前路由完全匹配为 /events/ 时，才会附加 className selected */}
<NavLink strict to="/events/" activeClassName="selected">Events</NavLink>
{/* 当前路由为 /courses ，并且 query 包含 name 时，会附加 className selected */}
<NavLink
to="/courses"
exact
activeClassName="selected"
isActive={(match, location) => {
if (!match) {
return false
}
return location.search.includes("name")
}}
>
Courses
</NavLink>
</>
)
}

Prompt#

用于在离开页面之前提示用户。

包含的属性有：

  • message : String/Function
  • when : Boolean
import {Prompt, useState} from 'concisejs'
function Example() {
const [formIsHalfFilledOut, setFormIsHalfFilledOut] = useState(false)
return (
<div>
{/* 用户离开页面时提示一个选择 */}
<Prompt message="确定离开吗？" />
{/* 用户离开页面时提示一个选择 */}
<Prompt
message={location => location.pathname !== "/" ? true : "您确定要跳转到首页吗？"}
/>
{/* 根据一个状态来确定用户离开页面时是否提示一个选择 */}
<Prompt when={formIsHalfFilledOut} message="系统可能不会保存您的更改，确定离开吗？" />
</div>
)
}

useHistory#

返回 history 对象。

import {useHistory} from 'concisejs'
function Example() {
const history = useHistory()
const onButtonClick = () => {
history.push('/home')
}
return (
<button type="button" onClick={onButtonClick}>
回到首页
</button>
)
}

useLocation#

返回代表当前 URL 的 location 对象。

import {useLocation} from 'concisejs'
function Example() {
const location = useLocation()
return (
<div>location: {location.pathname}</div>
)
}

useParams#

获取 params 对象。 params 对象为动态路由（例如： /users/:id ）里的参数键值对。

import {useParams} from 'concisejs'
function Example() {
const params = useParams()
// 对于 /users/:id 路由来说， params = {id: 值}
const {id} = params
return <div>id: {id}</div>
}

useRouteMatch#

获取当前路由的匹配信息。

import {useRouteMatch} from 'concisejs'
function Example() {
const match = useRouteMatch()
return <div>match: {JSON.stringify(match)}</div>
}
